  <behavior>
    - Hook signature: `useRealtimeWithFallback({ channelName, onPayload, subscribe, poll, pollIntervalMs = 10000 })`.
      - `channelName`: string (for React key stability only)
      - `onPayload`: (payload) => void
      - `subscribe(channelName, onPayload, onStatus)`: fn returning a cleanup fn. Caller builds the actual supabase.channel() inside this so the hook stays decoupled from any particular filter.
      - `poll()`: async fn that refetches the data (hook calls this on fallback trigger). Polling results are the caller's responsibility to merge — the hook just calls `poll()`.
    - Internal state: `isPolling: boolean`. Starts false. Flips true when `onStatus('CHANNEL_ERROR')` or `onStatus('TIMED_OUT')` fires. Flips false on successful `onStatus('SUBSCRIBED')` event that follows an error.
    - While `isPolling === true`: `setInterval(poll, pollIntervalMs)`. Clear on unmount or when `isPolling` flips back to false.
    - Cleanup on unmount runs BOTH the subscribe's returned cleanup fn AND clears any polling interval.
  </behavior>
  <action>
Create `src/lib/realtime/with-polling-fallback.ts`:

```ts
"use client";
import { useEffect, useRef, useState } from "react";

export type RealtimeStatus = "SUBSCRIBED" | "TIMED_OUT" | "CHANNEL_ERROR" | "CLOSED";

interface Params {
  channelName: string;
  subscribe: (onStatus: (status: RealtimeStatus) => void) => () => void;
  poll: () => void | Promise<void>;
  pollIntervalMs?: number;
}

export function useRealtimeWithFallback({ channelName, subscribe, poll, pollIntervalMs = 10_000 }: Params) {
  const [isPolling, setIsPolling] = useState(false);
  const pollRef = useRef(poll);
  useEffect(() => { pollRef.current = poll; }, [poll]);

  useEffect(() => {
    const cleanup = subscribe((status) => {
      if (status === "CHANNEL_ERROR" || status === "TIMED_OUT") {
        setIsPolling(true);
      } else if (status === "SUBSCRIBED") {
        setIsPolling(false);
      }
    });
    return cleanup;
    // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [channelName]);

  useEffect(() => {
    if (!isPolling) return;
    const id = setInterval(() => { void pollRef.current(); }, pollIntervalMs);
    return () => clearInterval(id);
  }, [isPolling, pollIntervalMs]);

  return { isPolling };
}
```

6. Poll callback receives updated ref after rerender (ref-based, not stale closure)
