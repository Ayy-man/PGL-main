```tsx
const handleDismiss = async (searchId: string, apolloPersonId: string, displayName: string) => {
  const previousProspects = prospects;
  setProspects(prev => prev.filter(p => p.apollo_person_id !== apolloPersonId));

  const handleUndo = async () => {
    setProspects(previousProspects); // restore UI
    // Fire undo on the server (card was already marked dismissed there)
    await fetch(`/api/search/${searchId}/dismiss`, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ action: "undo", apolloPersonIds: [apolloPersonId] }),
    });
  };

  toast({
    title: `Dismissed ${displayName}`,
    action: <ToastAction altText="Undo" onClick={handleUndo}>Undo</ToastAction>,
  });

  const res = await fetch(`/api/search/${searchId}/dismiss`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ action: "dismiss", apolloPersonIds: [apolloPersonId] }),
  });
  if (!res.ok) {
    setProspects(previousProspects);
    const err = await res.json().catch(() => ({}));
    toast({ title: "Dismiss failed", description: err.error || "Could not dismiss.", variant: "destructive" });
  }
};
```
