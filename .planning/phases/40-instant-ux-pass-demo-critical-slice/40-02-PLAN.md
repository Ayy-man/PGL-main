```ts
import type { RealtimePostgresChangesPayload } from "@supabase/supabase-js";

export type SavedSearchProspectRow = {
  id: string;
  prospect_id: string;
  enrichment_status: string | null;
  [key: string]: unknown;
};

export function shouldApplyUpdate(
  payload: RealtimePostgresChangesPayload<SavedSearchProspectRow>,
  visibleIds: Set<string>
): boolean {
  const row = payload.new as SavedSearchProspectRow | undefined;
  if (!row || !row.id) return false;
  return visibleIds.has(row.id);
}

export function reduceSavedSearchPayload<T extends { id: string; enrichment_status?: string | null }>(
  payload: RealtimePostgresChangesPayload<SavedSearchProspectRow>,
  currentMembers: T[]
): T[] {
  const row = payload.new as SavedSearchProspectRow | undefined;
  if (!row || !row.id || !row.enrichment_status) return currentMembers;
  const idx = currentMembers.findIndex(m => m.id === row.id);
  if (idx === -1) return currentMembers;
  if (currentMembers[idx].enrichment_status === row.enrichment_status) return currentMembers;
  const next = currentMembers.slice();
  next[idx] = { ...next[idx], enrichment_status: row.enrichment_status };
  return next;
}
```
