  if (!enrichingId) return;
  const channel = supabase
    .channel(`prospect-re-enrich-${enrichingId}`)
    .on('postgres_changes', { event: 'UPDATE', schema: 'public', table: 'prospects', filter: `id=eq.${enrichingId}` },

        const row = payload.new as any;
        if (row.enrichment_status === 'complete' || row.enrichment_status === 'failed') {
          setEnrichingId(null);
        }
      })
    .subscribe();
  return () => { supabase.removeChannel(channel); };
}, [enrichingId]);

Count-up helper (inline, no new deps):
```typescript
function useCountUp(target: number, duration = 600): number {
  const [value, setValue] = useState(0);
  useEffect(() => {
    const start = Date.now();
    let raf = 0;
    const tick = () => {
      const elapsed = Date.now() - start;
      if (elapsed >= duration) { setValue(target); return; }
      setValue(Math.floor((elapsed / duration) * target));
      raf = requestAnimationFrame(tick);
    };
    raf = requestAnimationFrame(tick);
    return () => cancelAnimationFrame(raf);
  }, [target, duration]);
  return value;
}
```
