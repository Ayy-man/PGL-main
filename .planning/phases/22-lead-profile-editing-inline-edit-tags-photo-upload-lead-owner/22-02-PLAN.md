    ```typescript
    /**
     * Field resolution for manual override pattern.
     * Display logic: manual_value ?? enriched_value ?? null
     */
    export function resolveField<T>(
      manual: T | null | undefined,
      enriched: T | null | undefined
    ): T | null {
      return manual ?? enriched ?? null;
    }

    /**
     * Check if a field has been manually overridden.
     * Used to show "edited" indicator with original value tooltip.
     */
    export function isOverridden(
      manual: string | null | undefined
    ): boolean {
      return manual !== null && manual !== undefined && manual !== '';
    }
    ```

    2. Create `src/components/prospect/inline-edit-field.tsx` as a "use client" component:

    Props interface:
    ```typescript
    interface InlineEditFieldProps {
      value: string | null;
      originalValue?: string | null;  // enriched value, shown in tooltip when overridden
      onSave: (newValue: string | null) => Promise<void>;
      placeholder?: string;
      displayClassName?: string;
      isEditable?: boolean;  // gated by RBAC canEdit
      label?: string;        // for aria-label
      isOverridden?: boolean; // shows "edited" indicator
      type?: 'text' | 'email' | 'url' | 'tel';
    }
    ```

    State machine: idle -> editing -> saving -> idle (with success flash)
                                              -> editing (on error, stay editable)

    Implementation details:
    - Container div with `className="group relative"` for CSS hover
    - In idle mode: display the value as a `<span>` with the pencil icon (`Pencil` from lucide-react) using `opacity-0 group-hover:opacity-100 transition-opacity` (CSS hover, NOT onMouseEnter)
    - When `isEditable` is false: render just the `<span>` with no pencil icon, no click handler
    - When `isOverridden` is true: show a small amber dot indicator and a `title` attribute on the value showing `"Original: ${originalValue}"`
    - Click pencil or double-click value: enter editing mode
    - In editing mode: render an `<input>` element with matching styling. Auto-focus on mount via `useRef` + `useEffect`. Value controlled via `useState`.
    - Enter key or click checkmark (`Check` icon): call `onSave(newValue)`, set saving state, show brief gold ring flash (`ring-1 ring-[var(--gold-primary)]`) for 500ms on success via setTimeout, return to idle
    - Escape key or click X (`X` icon): revert to previous value, return to idle
    - Optimistic UI: update displayed value immediately. On error in onSave, revert to previous value and stay in editing mode.
    - Empty string input: save as `null` (clear the override)
    - Gold flash: after successful save, add class `ring-1 ring-[var(--gold-primary)] transition-all duration-500` then remove after 500ms
    - All icons: 16px (`w-4 h-4`), muted color (`text-muted-foreground`), pencil button has `cursor-pointer`
    - Use `aria-label={label}` on the input for accessibility
  </action>
