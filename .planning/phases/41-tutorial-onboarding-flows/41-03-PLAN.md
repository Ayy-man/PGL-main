Create `src/lib/onboarding/video-url.ts`:
```typescript
export type VideoUrlResolution =
  | { kind: "url"; url: string }
  | { kind: "missing" }
  | { kind: "invalid" };

export function resolveVideoUrl(
  env: Partial<Record<string, string | undefined>>
): VideoUrlResolution {
  const raw = env.NEXT_PUBLIC_PGL_INTRO_VIDEO_URL;
  if (!raw || raw.trim() === "") return { kind: "missing" };
  try {
    const u = new URL(raw);
    if (u.protocol !== "https:" && u.protocol !== "http:") {
      return { kind: "invalid" };
    }
    return { kind: "url", url: raw };
  } catch {
    return { kind: "invalid" };
  }
}

export function isEmbeddableVideoUrl(url: string): boolean {
  try {
    const u = new URL(url);
    if (u.protocol !== "https:" && u.protocol !== "http:") return false;
    // Known hosts + direct video extensions
    const host = u.hostname.toLowerCase();
    if (host.includes("loom.com")) return true;
    if (host.includes("youtube.com") || host.includes("youtu.be")) return true;
    if (host.includes("vimeo.com")) return true;
    if (/\.(mp4|webm|mov)$/i.test(u.pathname)) return true;
    return false;
  } catch {
    return false;
  }
}
```

import { describe, it, expect } from "vitest";
import { resolveVideoUrl, isEmbeddableVideoUrl } from "../video-url";

describe("resolveVideoUrl", () => {
  it("returns url when env set", () => {
    const r = resolveVideoUrl({
      NEXT_PUBLIC_PGL_INTRO_VIDEO_URL: "https://loom.com/share/abc",
    });
    expect(r).toEqual({ kind: "url", url: "https://loom.com/share/abc" });
  });
  it("returns missing when env unset", () => {
    expect(resolveVideoUrl({}).kind).toBe("missing");
  });
  it("treats empty string as missing", () => {
    expect(
      resolveVideoUrl({ NEXT_PUBLIC_PGL_INTRO_VIDEO_URL: "" }).kind
    ).toBe("missing");
    expect(
      resolveVideoUrl({ NEXT_PUBLIC_PGL_INTRO_VIDEO_URL: "   " }).kind
    ).toBe("missing");
  });
  it("returns invalid for non-URL strings", () => {
    expect(
      resolveVideoUrl({ NEXT_PUBLIC_PGL_INTRO_VIDEO_URL: "not-a-url" }).kind
    ).toBe("invalid");
  });
});

describe("isEmbeddableVideoUrl", () => {
  it("accepts loom", () => {
    expect(isEmbeddableVideoUrl("https://loom.com/share/abc")).toBe(true);
  });
  it("accepts youtube", () => {
    expect(isEmbeddableVideoUrl("https://youtube.com/watch?v=abc")).toBe(true);
    expect(isEmbeddableVideoUrl("https://youtu.be/abc")).toBe(true);
  });
  it("accepts vimeo", () => {
    expect(isEmbeddableVideoUrl("https://vimeo.com/12345")).toBe(true);
  });
  it("accepts direct mp4/webm/mov", () => {
    expect(isEmbeddableVideoUrl("https://cdn.example.com/a.mp4")).toBe(true);
    expect(isEmbeddableVideoUrl("https://cdn.example.com/a.webm")).toBe(true);
  });
  it("rejects non-http(s)", () => {
    expect(isEmbeddableVideoUrl("ftp://example.com/video")).toBe(false);
  });
  it("rejects unparseable", () => {
    expect(isEmbeddableVideoUrl("not a url")).toBe(false);
  });
});

2. Create `src/components/layout/help-menu.tsx`:
```typescript
"use client";

import * as React from "react";
import { useRouter } from "next/navigation";
import { CircleHelp, PlayCircle, RotateCcw, Flag } from "lucide-react";
import {
  DropdownMenu,
  DropdownMenuTrigger,
  DropdownMenuContent,
  DropdownMenuItem,
} from "@/components/ui/dropdown-menu";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
} from "@/components/ui/dialog";
import { Button } from "@/components/ui/button";
import { updateOnboardingState } from "@/app/actions/onboarding-state";
import { resolveVideoUrl, pickVideoRenderer } from "@/lib/onboarding/video-url";
import { ReportIssueDialog } from "@/components/issues/report-issue-dialog";
import { captureScreenshot } from "@/lib/issues/capture-screenshot";

export function HelpMenu({ orgId }: { orgId?: string }) {
  const router = useRouter();
  const [videoOpen, setVideoOpen] = React.useState(false);
  const [issueOpen, setIssueOpen] = React.useState(false);
  const [issueBlob, setIssueBlob] = React.useState<Blob | null>(null);
  const [replaying, setReplaying] = React.useState(false);

  const videoRes = resolveVideoUrl({
    NEXT_PUBLIC_PGL_INTRO_VIDEO_URL:
      process.env.NEXT_PUBLIC_PGL_INTRO_VIDEO_URL,
  });

  const handleReplay = async () => {
    setReplaying(true);
    const res = await updateOnboardingState({ tour_completed: false });
    if (res.ok) {
      router.refresh();
    }
    setReplaying(false);
  };

  const handleReportIssue = async () => {
    const blob = await captureScreenshot().catch(() => null);
    setIssueBlob(blob);
    setIssueOpen(true);
  };

  return (
    <>
      <DropdownMenu>
        <DropdownMenuTrigger asChild>
          <Button
            variant="ghost"
            size="icon"
            aria-label="Help"
            className="h-8 w-8"
          >
            <CircleHelp className="h-4 w-4" />
          </Button>
        </DropdownMenuTrigger>
        <DropdownMenuContent align="end">
          <DropdownMenuItem onSelect={() => setVideoOpen(true)}>
            <PlayCircle className="mr-2 h-4 w-4" />
            Watch intro video
          </DropdownMenuItem>
          <DropdownMenuItem
            onSelect={handleReplay}
            disabled={replaying}
          >
            <RotateCcw className="mr-2 h-4 w-4" />
            {replaying ? "Resetting…" : "Replay product tour"}
          </DropdownMenuItem>
          <DropdownMenuItem onSelect={handleReportIssue}>
            <Flag className="mr-2 h-4 w-4" />
            Report an issue
          </DropdownMenuItem>
        </DropdownMenuContent>
      </DropdownMenu>

      <Dialog open={videoOpen} onOpenChange={setVideoOpen}>
        <DialogContent className="max-w-3xl">
          <DialogHeader>
            <DialogTitle>Welcome to Phronesis</DialogTitle>
          </DialogHeader>
          {videoRes.kind === "url" ? (
            pickVideoRenderer(videoRes.url) === "iframe" ? (
              <div className="aspect-video w-full overflow-hidden rounded-md">
                <iframe
                  src={videoRes.url}
                  title="Phronesis intro video"
                  className="h-full w-full"
                  allow="fullscreen; accelerometer; autoplay; clipboard-write; encrypted-media; gyroscope; picture-in-picture"
                />
              </div>
            ) : (
              <video
                src={videoRes.url}
                controls
                className="aspect-video w-full rounded-md"
              />
            )
          ) : (
            <div className="py-10 text-center text-sm text-muted-foreground">
              Video coming soon — check back after the demo.
            </div>
          )}
        </DialogContent>
      </Dialog>

      <ReportIssueDialog
        open={issueOpen}
        onOpenChange={(next) => {
          setIssueOpen(next);
          if (!next) setIssueBlob(null);
        }}
        target={{ kind: "help-menu", orgId }}
        preCapturedScreenshot={issueBlob}
      />
    </>
  );
}
```

NOTE on `target={{ kind: "help-menu", orgId }}`: match the existing `ReportTarget` union in `src/lib/issues/capture-context.ts`. If that union doesn't have a `"help-menu"` variant, use the closest existing variant (e.g., `{ kind: "generic", orgId }`) — do NOT add new variants in this phase.
