  invite_team: boolean;

export interface OnboardingState {
  tour_completed: boolean;
  tour_skipped_at?: string;
  tour_completed_at?: string;
  admin_checklist: AdminChecklistState;

export const DEFAULT_ONBOARDING_STATE: OnboardingState = {
  tour_completed: false,
  admin_checklist: {
    invite_team: false,
    upload_logo: false,
    pick_theme: false,
    create_first_persona: false,
  },
};

export type OnboardingStatePartial = {

  tour_skipped_at?: string;
  tour_completed_at?: string;

};

```typescript
"use server";

import {
  DEFAULT_ONBOARDING_STATE,
  type OnboardingState,
  type OnboardingStatePartial,
} from "@/types/onboarding";

// Pure helper — safe to unit-test, no side effects.
// NOTE: must stay export-able for Vitest import. Server Actions file with
// "use server" can still export non-async helpers in Next 14.
export function mergeOnboardingState(
  current: OnboardingState | null | undefined,
  partial: OnboardingStatePartial
): OnboardingState {
  const base = current ?? DEFAULT_ONBOARDING_STATE;
  const allowedTop: (keyof OnboardingStatePartial)[] = [
    "tour_completed",
    "tour_skipped_at",
    "tour_completed_at",
    "admin_checklist",
  ];
  const filtered: OnboardingStatePartial = {};
  for (const k of allowedTop) {
    if (k in partial) (filtered as Record<string, unknown>)[k] = partial[k];
  }
  return {
    ...base,
    ...filtered,
    admin_checklist: {
      ...base.admin_checklist,
      ...(filtered.admin_checklist ?? {}),
    },
  };
}
```

NOTE on `"use server"`: Next 14 forbids exporting non-async values from a `"use server"` module. Work around this by putting `mergeOnboardingState` in `src/lib/onboarding/merge-state.ts` (plain module) and re-importing from the actions file. Adjust paths accordingly — the pure helper ends up at `src/lib/onboarding/merge-state.ts`, the Server Action stays at `src/app/actions/onboarding-state.ts`. Update the `files_modified` / test-import path as needed.

Why we avoid spreading the raw `partial`: a buggy/malicious caller could drop a key like `tenant_id` or `role` into the object; filtering against `allowedTop` prevents that leak through to the later `auth.admin.updateUserById` call.

Append to `src/app/actions/onboarding-state.ts`:
```typescript
import { createClient } from "@/lib/supabase/server";
import { createAdminClient } from "@/lib/supabase/admin";
import { mergeOnboardingState } from "@/lib/onboarding/merge-state";
import type {
  OnboardingState,
  OnboardingStatePartial,
} from "@/types/onboarding";

export type UpdateOnboardingStateResult =
  | { ok: true; state: OnboardingState }
  | { ok: false; error: string };

export async function updateOnboardingState(
  partial: OnboardingStatePartial
): Promise<UpdateOnboardingStateResult> {
  const supabase = await createClient();
  const {
    data: { user },
    error: authError,
  } = await supabase.auth.getUser();

  if (authError || !user) {
    return { ok: false, error: "Not authenticated" };
  }

  const current =
    (user.app_metadata?.onboarding_state as OnboardingState | undefined) ??
    undefined;
  const next = mergeOnboardingState(current, partial);

  const admin = createAdminClient();
  const { error: updateError } = await admin.auth.admin.updateUserById(
    user.id,
    {
      app_metadata: {
        ...user.app_metadata,
        onboarding_state: next,
      },
    }
  );

  if (updateError) {
    console.error("[updateOnboardingState] Failed:", updateError);
    return { ok: false, error: "Failed to update onboarding state" };
  }

  return { ok: true, state: next };
}
```

import { describe, it, expect, vi, beforeEach } from "vitest";
import {
  DEFAULT_ONBOARDING_STATE,
  type OnboardingState,
} from "@/types/onboarding";
import { mergeOnboardingState } from "@/lib/onboarding/merge-state";

// --- mocks ---
const mockGetUser = vi.fn();
vi.mock("@/lib/supabase/server", () => ({
  createClient: async () => ({ auth: { getUser: mockGetUser } }),
}));

const mockUpdateUserById = vi.fn();
vi.mock("@/lib/supabase/admin", () => ({
  createAdminClient: () => ({
    auth: { admin: { updateUserById: mockUpdateUserById } },
  }),
}));

// --- Import SUT after mocks ---
import { updateOnboardingState } from "../onboarding-state";

describe("mergeOnboardingState", () => {
  it("applies partial over DEFAULT when current is undefined", () => {
    const next = mergeOnboardingState(undefined, { tour_completed: true });
    expect(next.tour_completed).toBe(true);
    expect(next.admin_checklist.invite_team).toBe(false);
  });

  it("deep-merges admin_checklist without clobbering other keys", () => {
    const current: OnboardingState = {
      tour_completed: true,
      admin_checklist: {
        invite_team: false,
        upload_logo: true,
        pick_theme: false,
        create_first_persona: false,
      },
    };
    const next = mergeOnboardingState(current, {
      admin_checklist: { invite_team: true },
    });
    expect(next.admin_checklist).toEqual({
      invite_team: true,
      upload_logo: true,
      pick_theme: false,
      create_first_persona: false,
    });
    expect(next.tour_completed).toBe(true);
  });

  it("does not mutate frozen inputs", () => {
    const current = Object.freeze({
      ...DEFAULT_ONBOARDING_STATE,
      admin_checklist: Object.freeze({ ...DEFAULT_ONBOARDING_STATE.admin_checklist }),
    }) as OnboardingState;
    expect(() =>
      mergeOnboardingState(current, { tour_completed: true })
    ).not.toThrow();
  });

  it("filters unknown keys out of partial", () => {
    const next = mergeOnboardingState(undefined, {
      tour_completed: true,
      // @ts-expect-error — intentional unknown key for defensive filter
      role: "super_admin",
    });
    expect((next as Record<string, unknown>).role).toBeUndefined();
  });
});

describe("updateOnboardingState", () => {
  beforeEach(() => {
    vi.clearAllMocks();
  });

  it("returns { ok:false, error } when unauthenticated", async () => {
    mockGetUser.mockResolvedValue({ data: { user: null }, error: null });
    const res = await updateOnboardingState({ tour_completed: true });
    expect(res).toEqual({ ok: false, error: "Not authenticated" });
    expect(mockUpdateUserById).not.toHaveBeenCalled();
  });

  it("preserves other app_metadata keys on write", async () => {
    mockGetUser.mockResolvedValue({
      data: {
        user: {
          id: "user-1",
          app_metadata: {
            tenant_id: "t-1",
            role: "agent",
            onboarding_completed: true,
          },
        },
      },
      error: null,
    });
    mockUpdateUserById.mockResolvedValue({ error: null });

    const res = await updateOnboardingState({ tour_completed: true });
    expect(res.ok).toBe(true);
    const [userId, payload] = mockUpdateUserById.mock.calls[0];
    expect(userId).toBe("user-1");
    expect(payload.app_metadata.tenant_id).toBe("t-1");
    expect(payload.app_metadata.role).toBe("agent");
    expect(payload.app_metadata.onboarding_completed).toBe(true);
    expect(payload.app_metadata.onboarding_state.tour_completed).toBe(true);
  });

  it("returns { ok:false, error } on admin update failure", async () => {
    mockGetUser.mockResolvedValue({
      data: { user: { id: "u", app_metadata: {} } },
      error: null,
    });
    mockUpdateUserById.mockResolvedValue({ error: { message: "boom" } });

    const res = await updateOnboardingState({ tour_completed: true });
    expect(res).toEqual({
      ok: false,
      error: "Failed to update onboarding state",
    });
  });
});
