    This is the core streaming endpoint. Uses Vercel AI SDK v6 pattern.

    ```typescript
    import { createUIMessageStream, createUIMessageStreamResponse } from "ai";
    import { createClient } from "@/lib/supabase/server";
    import { createAdminClient } from "@/lib/supabase/admin";
    import { searchExaForResearch } from "@/lib/research/exa-search";
    import { digestForScrapbook } from "@/lib/research/research-digest";
    import { researchRateLimiter } from "@/lib/research/research-rate-limit";
    import { chatCompletion } from "@/lib/ai/openrouter";
    import { logProspectActivity } from "@/lib/activity";
    import { NextResponse } from "next/server";
    ```

    **POST handler logic:**
    1. Auth: `createClient()` -> `getUser()` -> extract user_id, tenant_id from app_metadata
    2. Parse body: `{ query: string, session_id?: string }` validated with zod
    3. Rate limit: `await researchRateLimiter.limit(\`research:\${tenantId}\`)` — if `!success`, return 429 JSON: `{ error: "Daily research limit reached. Resets at midnight UTC." }`
    4. Fetch prospect from DB (need first_name, last_name, company for context): `admin.from("prospects").select("first_name, last_name, full_name, company, title").eq("id", prospectId).single()`
    5. Create or get session:
       - If `session_id` provided: use it
       - If null: insert into `research_sessions` with prospect_id, tenant_id, user_id, return new id
    6. Insert user message into `research_messages` with role="user", content=query, session_id
    7. Stream response using `createUIMessageStream`:

    **CRITICAL — Standardized SSE type names (must match Plan 04 client parser exactly):**
    The stream MUST use the following type names, which match the AI SDK v6 custom data part convention from RESEARCH.md:
    - `"data-reasoning"` — query reformulation status
    - `"data-tool"` — Exa search tool call status (uses `id` field for reconciliation)
    - `"data-shimmer"` — shimmer loading indicator toggle
    - `"data-card"` — individual result card (one per card)
    - `"data-sources"` — source URLs list

    ```typescript
    const stream = createUIMessageStream({
      execute: async ({ writer }) => {
        // Phase 1: Reasoning — query reformulation
        writer.write({
          type: "data-reasoning",
          data: { status: "reformulating", query },
          transient: true,
        });

        // Reformulate query with fast Claude call
        const reformulatedQuery = await chatCompletion(
          "You are a search query reformulator. Given a user question about a specific person, create an optimal web search query. Return ONLY the search query string, nothing else.",
          `Person: ${prospect.full_name}, ${prospect.title ?? ""} at ${prospect.company ?? ""}.\nUser question: ${query}`,
          100
        );

        writer.write({
          type: "data-reasoning",
          data: { status: "complete", reformulated: reformulatedQuery },
          transient: true,
        });

        // Phase 2: Tool call — Exa search
        writer.write({
          type: "data-tool",
          id: "exa-search",
          data: { status: "running", query: reformulatedQuery },
        });

        const exaResults = await searchExaForResearch(reformulatedQuery);

        writer.write({
          type: "data-tool",
          id: "exa-search",
          data: { status: "completed", count: exaResults.length },
        });

        // Phase 3: Shimmer then cards
        writer.write({
          type: "data-shimmer",
          data: { active: true },
          transient: true,
        });

        const cards = await digestForScrapbook(
          prospect.full_name,
          prospect.company ?? "",
          query,
          exaResults
        );

        writer.write({
          type: "data-shimmer",
          data: { active: false },
          transient: true,
        });

        // Stream cards one by one via data parts
        for (const card of cards) {
          writer.write({
            type: "data-card",
            data: card,
          });
        }

        // Phase 4: Sources
        writer.write({
          type: "data-sources",
          data: { urls: exaResults.map((r) => r.url).filter(Boolean) },
        });

        // Save assistant message with result_cards to DB
        await admin.from("research_messages").insert({
          session_id: sessionId,
          tenant_id: tenantId,
          role: "assistant",
          content: cards.length > 0
            ? `Found ${cards.length} results for: ${query}`
            : `No relevant results found for: ${query}`,
          result_cards: cards,
          metadata: {
            reformulated_query: reformulatedQuery,
            exa_result_count: exaResults.length,
            card_count: cards.length,
          },
        });

        // Update session updated_at
        await admin.from("research_sessions")
          .update({ updated_at: new Date().toISOString() })
          .eq("id", sessionId);
      },
    });
    ```

    8. Log activity (fire-and-forget after stream starts). Per CONTEXT.md locked decision, search activity uses eventType `"exa_updated"` with metadata `{ source: "research_scrapbook", query, results: count }`:
    ```typescript
    logProspectActivity({
      prospectId,
      tenantId,
      userId: user.id,
      category: "data",
      eventType: "exa_updated",
      title: "Research scrapbook search",
      metadata: { source: "research_scrapbook", query, results: cards.length },
    });
    ```

    9. Return: `return createUIMessageStreamResponse({ stream });`

    **IMPORTANT AI SDK v6 notes:**
    - Use `createUIMessageStream` NOT `StreamData` (removed in v6)
    - Use `createUIMessageStreamResponse` NOT `toDataStreamResponse`
    - The `writer.write()` accepts typed data parts with `type` field
    - For custom data, prefix type with `data-` (e.g., `data-card`, `data-sources`)
    - `transient: true` on data parts means they appear in `onData` callback but NOT in `message.parts` (use for ephemeral status like reasoning/shimmer)
    - Non-transient data parts (like `data-card`) persist in `message.parts` for re-rendering
