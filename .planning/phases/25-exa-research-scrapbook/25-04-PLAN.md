    **CRITICAL: Full SSE stream parsing implementation.**

    The server (Plan 03) uses `createUIMessageStream` which produces an SSE-formatted stream. The client MUST fully parse this stream using a ReadableStream reader loop. The stream format is newline-delimited: each line prefixed with a type code and colon, followed by JSON data.

    ```typescript
    const handleSend = async (query: string) => {
      setIsSearching(true);
      setStreamPhase("reasoning");
      setSuggestions([]); // hide suggestions after first query

      // Add user message to local state
      setMessages(prev => [...prev, { role: "user" as const, content: query }]);

      const cards: ScrapbookCard[] = [];
      let sources: string[] = [];

      try {
        const response = await fetch(`/api/prospects/${prospectId}/research`, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ query, session_id: sessionId }),
        });

        if (response.status === 429) {
          setMessages(prev => [...prev, {
            role: "assistant" as const,
            content: "Daily research limit reached. Resets at midnight UTC.",
          }]);
          setIsSearching(false);
          setStreamPhase("idle");
          return;
        }

        if (!response.ok || !response.body) {
          throw new Error(`Research request failed: ${response.status}`);
        }

        // --- SSE stream parsing ---
        // createUIMessageStream produces lines in the AI SDK UI protocol format.
        // Each line has a type prefix code followed by a colon and JSON payload.
        // We read the stream as text, split on newlines, and parse each line.

        const reader = response.body.getReader();
        const decoder = new TextDecoder();
        let buffer = "";

        while (true) {
          const { done, value } = await reader.read();
          if (done) break;

          buffer += decoder.decode(value, { stream: true });

          // Process complete lines from the buffer
          const lines = buffer.split("\n");
          // Keep the last (potentially incomplete) line in the buffer
          buffer = lines.pop() ?? "";

          for (const line of lines) {
            const trimmed = line.trim();
            if (!trimmed) continue;

            // AI SDK UI message stream protocol uses typed lines.
            // Data parts arrive as lines with specific prefixes.
            // Parse each line to extract the type and JSON payload.
            //
            // The key data part types we expect (matching Plan 03 server):
            //   "data-reasoning" -> update streamPhase to "reasoning"
            //   "data-tool"      -> update streamPhase to "tool"
            //   "data-shimmer"   -> update streamPhase to "shimmer" or back to previous
            //   "data-card"      -> append ScrapbookCard to cards array + update currentCards
            //   "data-sources"   -> capture source URLs
            //
            // Implementation: try to JSON-parse each non-empty line.
            // The AI SDK protocol wraps data parts in a JSON envelope with
            // a "type" field. Parse the envelope, then branch on type.

            try {
              // Try parsing the line as JSON (some protocol lines are JSON objects)
              // For data parts in the AI SDK UI protocol, the format is:
              // a numeric prefix followed by ":" and JSON data
              const colonIndex = trimmed.indexOf(":");
              if (colonIndex === -1) continue;

              const prefix = trimmed.substring(0, colonIndex);
              const payload = trimmed.substring(colonIndex + 1);
              if (!payload) continue;

              const parsed = JSON.parse(payload);

              // AI SDK data parts use numeric prefix codes.
              // Custom data parts (our data-* types) come through as arrays
              // or objects depending on the SDK version.
              // Handle both: direct type field or wrapped in array.

              // Extract the type from the parsed data
              const partType = parsed?.type
                ?? (Array.isArray(parsed) && parsed[0]?.type);

              if (!partType) {
                // Could be a text delta or other SDK protocol message — skip
                continue;
              }

              const partData = parsed?.data
                ?? (Array.isArray(parsed) && parsed[0]?.data)
                ?? parsed;

              switch (partType) {
                case "data-reasoning": {
                  setStreamPhase("reasoning");
                  // partData has { status, query?, reformulated? }
                  // Could display reformulated query in reasoning UI
                  break;
                }
                case "data-tool": {
                  setStreamPhase("tool");
                  // partData has { status: "running"|"completed", count? }
                  break;
                }
                case "data-shimmer": {
                  if (partData?.active) {
                    setStreamPhase("shimmer");
                  }
                  // When active: false, wait for cards to arrive
                  break;
                }
                case "data-card": {
                  setStreamPhase("cards");
                  // partData IS the ScrapbookCard object
                  const card = partData as ScrapbookCard;
                  cards.push(card);
                  // Update state progressively so cards render one by one
                  setCurrentCards([...cards]);
                  break;
                }
                case "data-sources": {
                  setStreamPhase("sources");
                  sources = partData?.urls ?? [];
                  setSourceUrls(sources);
                  break;
                }
              }
            } catch {
              // Non-JSON line or protocol overhead — skip silently
              continue;
            }
          }
        }

        // Process any remaining buffer
        if (buffer.trim()) {
          try {
            const colonIndex = buffer.trim().indexOf(":");
            if (colonIndex !== -1) {
              const payload = buffer.trim().substring(colonIndex + 1);
              const parsed = JSON.parse(payload);
              const partType = parsed?.type ?? (Array.isArray(parsed) && parsed[0]?.type);
              const partData = parsed?.data ?? (Array.isArray(parsed) && parsed[0]?.data) ?? parsed;
              if (partType === "data-card") {
                cards.push(partData as ScrapbookCard);
                setCurrentCards([...cards]);
              } else if (partType === "data-sources") {
                sources = partData?.urls ?? [];
                setSourceUrls(sources);
              }
            }
          } catch {
            // ignore
          }
        }

        // After stream completes — add assistant message with all cards
        setStreamPhase("complete");
        setMessages(prev => [...prev, {
          role: "assistant" as const,
          content: cards.length > 0
            ? `Found ${cards.length} results`
            : "No relevant results found",
          cards,
        }]);

        // Extract session_id from response headers if this was a new session
        // (Plan 03 should set x-session-id header or include in stream)
        // For now, the session_id arrives implicitly — on next query the server
        // will reuse the same session if we pass it.

      } catch (err) {
        console.error("Research stream error:", err);
        setMessages(prev => [...prev, {
          role: "assistant" as const,
          content: "Research failed. Please try again.",
        }]);
      } finally {
        setIsSearching(false);
        setStreamPhase("idle");
        setCurrentCards([]);
      }
    };
    ```
